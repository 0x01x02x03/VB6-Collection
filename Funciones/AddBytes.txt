Private Function AddBytes(FilePath As String, NBytes As Long, Optional kb As Long = False)
If kb = True Then kb = 1024 Else kb = 1
Open FilePath For Binary As #1
Put #1, LOF(1) + NBytes * kb, CByte(&H0)
Close #1
End Function
